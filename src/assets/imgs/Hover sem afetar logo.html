⚙️ JavaScript

Cria um arquivo carousel.js dentro da pasta src/js/ e adiciona:

js
let currentSlide = 0;
const slides = document.querySelectorAll(".slide");
const dots = document.querySelectorAll(".dot");

function showSlide(index) {
    slides.forEach((slide, i) => {
        slide.classList.toggle("active", i === index);
        dots[i].classList.toggle("active", i === index);
    });
    currentSlide = index;
}

function nextSlide() {
    let next = (currentSlide + 1) % slides.length;
    showSlide(next);
}

function prevSlide() {
    let prev = (currentSlide - 1 + slides.length) % slides.length;
    showSlide(prev);
}

document.querySelector(".next").addEventListener("click", nextSlide);
document.querySelector(".prev").addEventListener("click", prevSlide);
dots.forEach((dot, i) => dot.addEventListener("click", () => showSlide(i)));

setInterval(nextSlide, 15000);

window.addEventListener("DOMContentLoaded", () => showSlide(0));

E no seu index.html, antes de fechar o </body>:

html
<script src="./src/js/carousel.js"></script>

Claro! Bem tranquilo de resolver.

O problema é que no seu CSS esse seletor:

css
.header-nav a:hover {
    border-bottom: 1px solid white;
}

aplica o hover em todos os <a>, incluindo o que tem a logo.

Solução:

Você pode mudar o CSS para aplicar só nos links que NÃO são a logo.

Um jeito fácil: usar :not() para excluir o link que contém o img#logo:

css
.header-nav a:not(:has(img)):hover {
    border-bottom: 1px solid white;
}

Esse :not(:has(img)) diz: "não aplicar o hover nos <a> que têm uma imagem dentro" — perfeito pro seu caso.

Resumo:

Troca isso no seu style.css:

css
.header-nav a:hover {

css
.header-nav a:not(:has(img)):hover {
    border-bottom: 1px solid white;
}
